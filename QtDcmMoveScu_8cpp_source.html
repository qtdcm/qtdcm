165  d->currentSerie = d->series.at ( i );

167 
168  if ( !serieDir.exists() ) {
169  QDir ( d->outputDir ).mkdir ( d->series.at ( i ) );
170  }
171 
172  d->outputDirectory = QString ( d->outputDir + QDir::separator() + d->currentSerie ).toUtf8().constData();
173 
174  if ( d->mode == IMPORT ) {
175  cond = this->move ( d->series.at ( i ) );
176  emit updateProgress ( ( int ) ( 100.0 * ( i+1 ) / d->series.size() ) );
177  d->progressTotal += d->step;
178  emit serieMoved ( serieDir.absolutePath(), d->series.at ( i ), i );
179  }
180  else {
181  cond = this->move ( d->imageId );
182  }

292  return cond;

294  }

303  }
304  }
305  }

307  qDebug() << "Protocol Error: Peer requested release (Aborting)";
308  cond = ASC_abortAssociation ( d->assoc );
309  ASC_dropNetwork ( &d->net );
310 
311  if ( cond.bad() ) {
312  qDebug() << "Association Abort Failed: " << DimseCondition::dump ( temp_str,cond ).c_str();
313  return cond;
314  }
315 
316  return cond;
317  }

320  ASC_abortAssociation ( d->assoc );
321  ASC_dropNetwork ( &d->net );
322 
323  return cond;
324  }

326  qDebug() << "Move SCU Failed: Aborting Association";
327  cond = ASC_abortAssociation ( d->assoc );

340 void QtDcmMoveScu::addOverrideKey ( const QString & key )
341 {
342  char * s = key.toUtf8().data();
343  unsigned int g = 0xffff;
344  unsigned int e = 0xffff;
345  int n = 0;
346  OFString dicName, valStr;
347  OFString msg;
348  char msg2[200];
349 
350  // try to parse group and element number
351  n = sscanf ( s, "%x,%x=", &g, &e );
352  OFString toParse = s;
353  size_t eqPos = toParse.find ( '=' );
354 
355  if ( n < 2 ) { // if at least no tag could be parsed
356  // if value is given, extract it (and extrect dictname)
357  if ( eqPos != OFString_npos ) {
358  dicName = toParse.substr ( 0, eqPos ).c_str();
359  valStr = toParse.substr ( eqPos + 1, toParse.length() );
360  }
361  else {
362  // no value given, just dictionary name
363  dicName = s; // only dictionary name given (without value)
364  }
365 
366  // try to lookup in dictionary
367  DcmTagKey key ( 0xffff, 0xffff );
368  const DcmDataDictionary& globalDataDict = dcmDataDict.rdlock();
369  const DcmDictEntry *dicent = globalDataDict.findEntry ( dicName.c_str() );
370 
371  dcmDataDict.unlock();
372 
373  if ( dicent != NULL ) {
374  // found dictionary name, copy group and element number
375  key = dicent->getKey();
376  g = key.getGroup();
377  e = key.getElement();
378  }
379  else {
380  // not found in dictionary
381  msg = "bad key format or dictionary name not found in dictionary: ";
382  msg += dicName;
383  qDebug() << QString ( msg.c_str() );
384  }
385  } // tag could be parsed, copy value if it exists
386  else {
387  if ( eqPos != OFString_npos )
388  valStr = toParse.substr ( eqPos + 1, toParse.length() );
389  }
390 
391  DcmTag tag ( g, e );
392  if ( tag.error() != EC_Normal ) {
393  sprintf ( msg2, "unknown tag: (%04x,%04x)", g, e );
394  qDebug() << QString ( msg2 );
395  }
396 
397  DcmElement *elem = newDicomElement ( tag );
398  if ( elem == NULL ) {
399  sprintf ( msg2, "cannot create element for tag: (%04x,%04x)", g, e );
400  qDebug() << QString ( msg2 );
401  }
402 
403  if ( valStr.length() > 0 ) {
404  if ( elem->putString ( valStr.c_str() ).bad() ) {
405  sprintf ( msg2, "cannot put tag value: (%04x,%04x)=\"", g, e );
406  msg = msg2;
407  msg += valStr;
408  msg += "\"";
409  qDebug() << QString ( msg.c_str() );
410  }
411  }
412 
413  if ( d->overrideKeys.insert ( elem, OFTrue ).bad() ) {
414  sprintf ( msg2, "cannot insert tag: (%04x,%04x)", g, e );
415  qDebug() << QString ( msg2 );
416  }
417 }

419 OFCondition QtDcmMoveScu::addPresentationContext ( T_ASC_Parameters *params, T_ASC_PresentationContextID pid, const char* abstractSyntax, E_TransferSyntax preferredTransferSyntax )
420 {
421  const char* transferSyntaxes[] = { NULL, NULL, NULL };
422  int numTransferSyntaxes = 0;
423 
424  switch ( preferredTransferSyntax )
425  {
426  case EXS_LittleEndianImplicit:
427  /* we only support Little Endian Implicit */
428  transferSyntaxes[0] = UID_LittleEndianImplicitTransferSyntax;
429  numTransferSyntaxes = 1;
430  break;
431 
432  case EXS_LittleEndianExplicit:
433  /* we prefer Little Endian Explicit */
434  transferSyntaxes[0] = UID_LittleEndianExplicitTransferSyntax;
435  transferSyntaxes[1] = UID_BigEndianExplicitTransferSyntax;
436  transferSyntaxes[2] = UID_LittleEndianImplicitTransferSyntax;
437  numTransferSyntaxes = 3;
438  break;
439 
440  case EXS_BigEndianExplicit:
441  /* we prefer Big Endian Explicit */
442  transferSyntaxes[0] = UID_BigEndianExplicitTransferSyntax;
443  transferSyntaxes[1] = UID_LittleEndianExplicitTransferSyntax;
444  transferSyntaxes[2] = UID_LittleEndianImplicitTransferSyntax;
445  numTransferSyntaxes = 3;
446  break;
447 
448  default:
449  /* We prefer explicit transfer syntaxes.
450  * If we are running on a Little Endian machine we prefer
451  * LittleEndianExplicitTransferSyntax to BigEndianTransferSyntax.
452  */
453 
454  if ( gLocalByteOrder == EBO_LittleEndian ) /* defined in dcxfer.h */ {
455  transferSyntaxes[0] = UID_LittleEndianExplicitTransferSyntax;
456  transferSyntaxes[1] = UID_BigEndianExplicitTransferSyntax;
457  }
458  else {
459  transferSyntaxes[0] = UID_BigEndianExplicitTransferSyntax;
460  transferSyntaxes[1] = UID_LittleEndianExplicitTransferSyntax;
461  }
462 
463  transferSyntaxes[2] = UID_LittleEndianImplicitTransferSyntax;
464 
465  numTransferSyntaxes = 3;
466  break;
467  }
468 
469  return ASC_addPresentationContext (
470  params, pid, abstractSyntax,
471  transferSyntaxes, numTransferSyntaxes );
472 }
